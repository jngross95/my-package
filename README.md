example (G H U : Prop)(h : G ∧ (H ∨ U)) : (G ∧ H) ∨ (G ∧ U) := by
have g := h.left


have f1:(H->(G ∧ H) ∨ (G ∧ U)) := λh:H => (or_inl (and_intro g h))

have f2:(U->(G ∧ H) ∨ (G ∧ U)) := λu:U => (or_inr (and_intro g u))

have z := or_elim h.right  f1 f2
exact z
